# Simple Map

See https://openlayers.org/en/latest/examples/simple.html

```tsx
import React from "react";
import "ol/ol.css";
import { Map } from "@react-ol/fiber";

export const Example = () => (
  <Map style={{ width: "100%", height: "640px" }}>
    <olView initialCenter={[0, 0]} initialZoom={2} />
    <olLayerTile>
      <olSourceOSM />
    </olLayerTile>
  </Map>
);
```
